## 1. 実現可能性評価

### 1.1 結論

**✅ 実現可能。既存システムの延長線上で構築できる。**

| 評価軸 | 評価 | 根拠 |
|---|---|---|
| 技術的実現性 | ◎ | 既存の recharts・Next.js・TypeScript 上に追加するだけ |
| データ構造の拡張性 | ◎ | QuarterData に施策IDを紐付けるだけで相関分析が可能 |
| AI分析の組み込み | ○ | governance-tool が同一パターンで実装済み（Gemini API） |
| 分析精度 | △ | ダミーデータ段階では推察ベース。実データ蓄積で精度向上 |
| 開発工数（概算） | ○ | Phase追加実装として 2〜3週間程度 |

### 1.2 実現できること・できないこと

**できること（定量・定性分析）**
- 施策の実施タイミングをチャート上に可視化（参照ライン）
- 施策実施前後の満足度スコア差分の自動計算
- 施策カテゴリと影響を受けたカテゴリの相関ヒートマップ
- AI（Gemini）による「施策→スコア変化」の推察コメント生成
- 複数施策の重複期間を考慮した影響度の整理

**できないこと（制約）**
- 因果関係の科学的証明（相関は示せるが因果は不明）
- 外部要因（景気・業界動向）との分離
- リアルタイムの回答収集（現状ダミーデータ運用）

```typescript
// 施策カテゴリ（満足度の10カテゴリと対応させる）
export type MeasureCategory =
  | "work_life_balance"   // ワークライフバランス改善施策
  | "career_growth"       // キャリア・育成施策
  | "relationships"       // 人間関係・チームビルディング
  | "fulfillment"         // やりがい・エンゲージメント
  | "compensation"        // 給与・報酬・福利厚生
  | "culture"             // 企業文化・ビジョン浸透
  | "management"          // マネジメント改善
  | "recognition"         // 評価制度・承認
  | "environment"         // 職場環境整備
  | "communication"       // 情報共有・コミュニケーション
  | "cross";              // 横断的施策（複数カテゴリに影響）

// 施策マスタ
export interface Measure {
  id: string;                    // UUID
  name: string;                  // "1on1制度導入"
  description: string;           // 施策の詳細説明
  category: MeasureCategory;     // 主たるカテゴリ
  relatedCategories: MeasureCategory[]; // 副次的に影響するカテゴリ
  startQuarter: QuarterKey;      // 開始四半期
  endQuarter: QuarterKey | null; // 終了四半期（null = 継続中）
  targetScore: number | null;    // 目標スコア（任意）
  targetTurnoverRate: number | null; // 目標離職率（任意）
  status: "planning" | "active" | "completed" | "suspended";
  owner: string;                 // 担当者名
  cost: "low" | "medium" | "high"; // コスト感
  createdAt: string;
}

// 施策効果スナップショット（四半期集計時に自動生成）
export interface MeasureEffect {
  measureId: string;
  quarter: QuarterKey;           // 評価した四半期
  baselineQuarter: QuarterKey;   // 比較基準四半期（施策開始前）

  // スコア変化
  overallScoreDiff: number;      // 総合スコア差分（+5 など）
  categoryDiffs: Record<string, number>; // カテゴリ別差分
  turnoverRateDiff: number;      // 離職率差分（-1.2% など）

  // 信頼度
  confidence: "low" | "medium" | "high"; // 他施策との重複考慮
  confoundingMeasures: string[]; // 同時期に実施した他施策ID

  // AI分析
  aiComment: string | null;      // Gemini による推察コメント
}

// AI分析レポート
export interface MeasureAnalysisReport {
  generatedAt: string;
  quarters: QuarterKey[];        // 分析対象四半期
  measures: Measure[];           // 分析対象施策

  // 総合評価
  overallAssessment: string;

  // 施策別評価
  measureInsights: {
    measureId: string;
    impact: "high" | "medium" | "low" | "uncertain";
    summary: string;             // "離職率の2.7pt低下に寄与した可能性"
    evidence: string;            // 根拠となる数値
  }[];

  // 推奨アクション
  recommendations: {
    priority: number;
    action: string;
    targetCategory: MeasureCategory;
    expectedEffect: string;
  }[];

  // リスク
  risks: string[];
}
```

## 5. 分析ロジック設計

### 5.1 施策前後差分の計算

```typescript
function calcMeasureEffect(
  measure: Measure,
  quarterData: QuarterData[]
): MeasureEffect {

  // 施策開始直前の四半期をベースライン
  const baselineIdx = quarterData.findIndex(
    q => q.quarter === measure.startQuarter
  ) - 1;

  // 評価四半期（施策開始後の最新四半期）
  const latestIdx = quarterData.length - 1;

  const baseline = quarterData[baselineIdx];
  const latest   = quarterData[latestIdx];

  // カテゴリ別差分
  const categoryDiffs = Object.fromEntries(
    Object.keys(baseline.categoryScores).map(cat => [
      cat,
      (latest.categoryScores[cat] ?? 0) - (baseline.categoryScores[cat] ?? 0)
    ])
  );

  // 同時期に実施された他施策（交絡因子）
  const confounding = quarterData
    .slice(baselineIdx + 1)
    .flatMap(q => q.activeMeasures)
    .filter(id => id !== measure.id);

  // 信頼度：交絡因子が多いほど低い
  const confidence =
    confounding.length === 0 ? "high" :
    confounding.length <= 2  ? "medium" : "low";

  return {
    measureId: measure.id,
    quarter: latest.quarter,
    baselineQuarter: baseline.quarter,
    overallScoreDiff: latest.totalScore - baseline.totalScore,
    categoryDiffs,
    turnoverRateDiff: latest.turnoverRate - baseline.turnoverRate,
    confidence,
    confoundingMeasures: [...new Set(confounding)],
    aiComment: null, // AI分析後に付与
  };
}
```

### 5.2 相関ヒートマップの生成

```typescript
// 施策カテゴリ × 満足度カテゴリ の変化率マトリクス
function buildCorrelationMatrix(
  effects: MeasureEffect[],
  measures: Measure[]
): Record<string, Record<string, number>> {

  return Object.fromEntries(
    measures.map(m => {
      const effect = effects.find(e => e.measureId === m.id);
      return [
        m.name,
        effect?.categoryDiffs ?? {}
      ];
    })
  );
}
```

### 5.3 離職率への影響推計（線形回帰）

```typescript
// 満足度スコアと離職率の単回帰
// y（離職率） = a × x（満足度） + b
function estimateTurnoverImpact(
  historicalData: QuarterData[],
  scoreDiff: number
): number {

  // 最小二乗法で係数を推定
  const n = historicalData.length;
  const xMean = avg(historicalData.map(d => d.totalScore));
  const yMean = avg(historicalData.map(d => d.turnoverRate));

  const a = historicalData.reduce((num, d) =>
    num + (d.totalScore - xMean) * (d.turnoverRate - yMean), 0
  ) / historicalData.reduce((den, d) =>
    den + Math.pow(d.totalScore - xMean, 2), 0
  );

  // 施策によるスコア改善が離職率に与える推計影響
  return a * scoreDiff; // 例: -0.08 * +9 = -0.72%
}
```
